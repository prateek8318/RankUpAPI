# ===== EC2 DEPLOYMENT COMMANDS =====
# Copy paste these commands one by one on EC2

# 1. Update System (Amazon Linux 2)
sudo yum update -y

# 2. Install Docker
sudo yum install -y docker
sudo systemctl start docker
sudo systemctl enable docker
sudo usermod -a -G docker ec2-user

# 3. Install Git
sudo yum install -y git

# 4. Logout and login again (for Docker group)
exit
# Then SSH again

# 5. Clone your repository
git clone https://github.com/yourusername/RankUpAPI.git
cd RankUpAPI

# 6. Setup environment file
cp .env.example .env

# 7. Edit environment file (IMPORTANT!)
nano .env
# Update these lines:
# MSSQL_SA_PASSWORD=YourStrongPassword123!
# GATEWAY_PORT=5087
# Uncomment production settings

# 8. Deploy microservices
docker compose -f docker/docker-compose.yml up --build -d

# 9. Check if everything is running
docker compose -f docker/docker-compose.yml ps

# 10. Test the API
curl http://localhost:5087/api/users

# 11. View logs if needed
docker compose -f docker/docker-compose.yml logs -f gateway

# ===== SUCCESS! YOUR API IS RUNNING =====
# Access: http://your-ec2-public-ip:5087
